// تهيئة حزمة Pi المبرمجية Pi.init({ version: "2.0", sandbox: true }); // دالة تسجيل الدخول التي يتم استدعاؤها عند الضغط على الزر async function authUser() { try { // طلب التوثيق الرسمي من متصفح Pi Browser const scopes = ['username', 'payments']; const auth = await Pi.authenticate(scopes, onIncompletePaymentFound); // إذا نجح تسجيل الدخول، يمكنك توجيه المستخدم لصفحة الحساب الداخلية alert(`أهلاً بك ${auth.user.username}`); window.location.href = "dashboard.html"; } catch (error) { console.error("فشل تسجيل الدخول:", error); alert("تأكد من فتح الرابط من داخل متصفح Pi Browser الرسمي."); } } function onIncompletePaymentFound(payment) { console.log("Incomplete payment found", payment); }
